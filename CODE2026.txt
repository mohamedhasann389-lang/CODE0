        <button class="color-button" id="reset-color">Reset Color</button>

    <script>
        // Get elements from the page
        const changeColorButton = document.getElementById('change-color');
        const resetColorButton = document.getElementById('reset-color');
        const currentColorSpan = document.getElementById('current-color');
        const body = document.body;
        
        // Array of colors to choose from
        const colors = [
            '#f0f8ff', '#ffe4e1', '#e6e6fa', '#f5f5dc', 
            '#f0fff0', '#fff0f5', '#f5fffa', '#fffacd',
            '#e0ffff', '#f8f8ff', '#fafad2', '#ffe4b5'
        ];
        
        // Function to generate a random color
        function getRandomColor() {
            const randomIndex = Math.floor(Math.random() * colors.length);
            return colors[randomIndex];
        }
        
        // Function to change the background color
        function changeBackgroundColor() {
            const newColor = getRandomColor();
            body.style.backgroundColor = newColor;
            currentColorSpan.textContent = newColor;
        }
        
        // Function to reset the background color
        function resetBackgroundColor() {
            body.style.backgroundColor = '#f0f8ff';
            currentColorSpan.textContent = '#f0f8ff';
        }
        
        // Add event listeners to buttons
        changeColorButton.addEventListener('click', changeBackgroundColor);
        resetColorButton.addEventListener('click', resetBackgroundColor);
    </script>
